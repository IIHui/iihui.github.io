int adjustByFirst(int *data, int start, int end) {
    int cur = data[start];
    int i = start, j = end;
    while (i < j) {
        while (i < j && data[j] >= cur) j--;
        if (i < j) {
            data[i] = data[j];

        }
    }

    return i;
}
